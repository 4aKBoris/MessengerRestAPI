package com.mpei

import org.junit.Test
import java.nio.charset.Charset
import java.security.MessageDigest

class MD5Digest {
    @Test
    fun digest() {
        val phoneNumber = "79255696418"
        val password = "123456"
        val str = "$phoneNumber:$RealM:$password"
        println(bytes(str = str).joinToString())
    }

    companion object {

        private const val RealM = "RestAPI"

        private fun bytes(str: String) =
            MessageDigest.getInstance("MD5").digest(str.toByteArray(charset = Charset.defaultCharset()))
    }
}

@ExperimentalPermissionsApi
@Composable
fun PermissionUI() {
    val context = LocalContext.current
    val permissionState = rememberPermissionState(permission = Manifest.permission.READ_SMS)
    var doNotShowMeRationale by rememberSaveable {
        mutableStateOf(false)
    }

    PermissionRequired(
        permissionState = permissionState,
        permissionNotGrantedContent = {
            if (doNotShowMeRationale) {
                Text(
                    text = "В разрешение  было оказано. " +
                            "Вы можете выдвать его в настройках."
                )
            } else {
                PermissionNotGrantedUI(
                    onYesClick = { permissionState.launchPermissionRequest() },
                    onCancelClick = { doNotShowMeRationale = true })
            }
        },
        permissionNotAvailableContent = {

            PermissionNotAvailableContent(onOpenSettingsClick = {
                context.openSettings()
            })
        },
        content = {
            Text(text = "Все разрешения выданы.")
        })
}
